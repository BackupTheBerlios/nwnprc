feat.2da, second sequential access (LIST GENERIC AND BONUS FEATS)

HASREQS = (MINATTACKBONUS<=AttackBonus 
	AND MINSTR<=Str ... AND MINCHA<=Cha 
	AND MINSPELLLVL<=spelllvl 
	AND MINLEVEL<=1 AND (MinLevelClass=Class OR MinLevelClass="") 
	AND (MAXLEVEL>=1 OR MAXLEVEL ="") 
	AND MINFORTSAVE<=FortSave 
	AND PREREQEPIC=0
)

WHERE (HASREQS) AND ((row=[all GrantedOnLevel=-1 from cls_feat]) OR ALLCLASSESCANUSE=1)
{
	row
    FEAT
    DESCRIPTION
    ALLCLASSESCANUSE
    PREREQFEAT1
    PREREQFEAT2
    OrReqFeat0
    OrReqFeat1
    OrReqFeat2
    OrReqFeat3
    OrReqFeat4
    SUCCESSOR
    MASTERFEAT
    REQSKILL
    ReqSkillMinRanks
    REQSKILL2
    ReqSkillMinRanks2
}
